df_all_days <- data.frame(ts_utc=seq(
  ymd_hms(paste0(date("2020-02-10"), " 0:00:00"), tz="UTC"),
  ymd_hms(paste0(date("2021-08-01"), " 23:59:59"), tz="UTC"), by="10 mins")) 

## add ind_min and day number to df_all_days
## ind_day number 20 is a leap year day
df_all_days <- df_all_days |> 
  mutate(ind_min=rep(1:144, distinct(df_all_days, date(ts_utc))|>nrow())) |>
  mutate(ind_day0=as.numeric(as.factor(format(ts_utc, tz="UTC", "%Y-%m-%d")))) |>
  mutate(ind_year0=as.numeric(as.factor(format(ts_utc, tz="UTC", "%Y")))) |>
  mutate(ind_day=yday(ymd(format(ts_utc, tz="UTC", "%Y-%m-%d"))))

## merge df and df_all_days include td01
dfnew <- df_all_days |> 
  left_join(dfa, by="ts_utc") |>
  rename(way="p76_median", temp="p1_median") |>
  group_by(ind_day0) |>
  mutate(td01 = seq(.5, 143.5, 1)) |>
  ungroup()

# include leap year day in February
dtai <- dfnew |>
  mutate(ind_day0=ifelse(ind_day0==20 & ind_year0==1, 19.9, ind_day0)) |>
  mutate(ind_day0=ifelse(ind_day0>20 & ind_year0==1, ind_day0-1, ind_day0)) |>
  mutate(ind_day0=ifelse(ind_year0==2, ind_day0-1, ind_day0))

# replace abnormal (wrong) way and temp data values with NA
dtai <- dtai |>
  mutate(way  = replace(way, ind_day0==382 & ind_min==116, NA),
         temp = replace(temp, ind_day0==29 & ind_min%in%c(45:52), NA),
         temp = replace(temp, ind_day0==190 & ind_min%in%c(71:72), NA),
         temp = replace(temp, ind_day0==191 & ind_min%in%c(75:76), NA),

Refit with functional random effects (lme and mgcv)

eq <- paste0("way~t2(", "td01, ind_day", ",bs = 'cc')", "+", 
             "s(", "temp", ", bs = 'cr')")
re <- paste0("~", paste((sprintf("ef1%02d", 1:fpcaRes1$npc)), collapse="+"),"+",
              paste((sprintf("ef2%02d", 1:fpcaRes2$npc)), collapse="+"), "-1")
gam6d <- gamm(as.formula(eq), data=dtaiB6, method="ML", random=list(ind_dayf=pdDiag(as.formula(re))),
                            control=list(msMaxIter=100, msMaxEval=1000))

re <- paste0("~", paste((sprintf("ef1%02d", 1:fpcaRes1$npc)), collapse="+"),"+",
              paste((sprintf("ef2%02d", 1:fpcaRes2$npc)), collapse="+"), "-1")
gam6d <- gamm(as.formula(eq), data=dtaiB6, method="ML", random=list(ind_dayf=pdDiag(as.formula(re))),
                            control=list(msMaxIter=100, msMaxEval=1000))
